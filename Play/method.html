1. push() : 배열의 끝에 요소 추가하기 ex let fruits = ["apple", "banana"]; fruits.push("orange"); console.log(fruits); 출력시 : "apple", "banana", "orange" 2. pop() : 배열의 마지막 요소 제거하기 ex let fruits = ["apple", "banana", "orange"]; let removed = fruits.pop(); console.log(removed); // "orange" console.log(fruits); // ["apple", "banana"] 3. shift() : 배열의 첫 번째 요소 제거하기 ex let fruits = ["apple", "banana", "orange"]; let removed = fruits.shift(); console.log(removed); // "apple" console.log(fruits); // "banana", "orange" 4. unshift() : 배열의 시작 부분에 요소 추가하기 ex let fruits = ["banana", "orange"]; fruits.unshift("apple"); console.log(fruits); "apple", "banana", "orange" 5. concat() : 배열 병합하기 ex let number1 = [1, 2, 3]; let number2 = [4, 5, 6]; let combined = numbers1.concat(number2); console.log(combined); // [1, 2, 3, 4, 5, 6]; 6. slice() : 배열 일부 추출하기 ex let fruits = ["apple", "banana", "orange", "kiwi"]; let citrus = fruits.sloce(1, 3); console.log(citrus);//"banana", "orange" 7. splice() : 배열 수정하기(요소추가/제거) ex let fruits = ["apple", "banana", "orange", "kiwi"]; fruits.splice(2, 1, "grape"); console.log(fruits); // "apple", "banana", "grape", "kiwi" 8. indexOf() : 요소의 첫 번째 인덱스 찾기 ex let fruits = ["apple", "banana", "orange", "banana"]; let index = fruits.indexOf("banana"); console.log(index); // 1 9. lastIndexOf() : 요소의 마지막 인덱스 찾기 ex let fruits = ["apple", "banana", "orange", "banana"]; let lastIndex = fruits.lastIndexOf("banana"); console.log(lastIndex); // 3 10. join() : 배열 요소 문자열로 결합하기 ex let fruits = ["apple", "banana", "orange"]; let result = fruits.join(", "); console.log(result); // "apple, banana, orange" 11. reverse() : 배열 순서 반전하기 let numbers = [1, 2, 3, 4, 5]; numbers.reverse(); console.log(numbers); // [5, 4, 3, 2, 1]
